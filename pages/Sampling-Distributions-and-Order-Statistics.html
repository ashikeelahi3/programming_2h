Problem 3: A random sample of size 4 is drawn from a population that has a uniform distribution on the interval (0, 5). The resulting order statistics are \(X_{1:4}, X_{2:4}, X_{3:4}\) and \(X_{4:4}\). Determine the cumulative distribution function (CDF) of the \(3^{rd}\) order statistic \((X_{3:4})\) and plot the CDF of the \(3^{rd}\) order statistic. Evaluate the probability \(P[X_{3:4} \geq 2]\)
Fr_n = function(x, r, n, a, b) {
  cdf_x = punif(x, min = a, max = b)
  sum_value = 0
  for(j in r:n) {
    sum_value = sum_value + choose(n, j) * cdf_x^j * (1-cdf_x)^(n-j)
  }
  return(sum_value)
}

Fr_n_eqn = function(x, r, n) {
  eqn = ""
  for(j in r:n) {
    term = paste0("(", n, "C", j, ")*F(x)^", j, "*[1-F(x)]^", n-j)
    if(j < n) {
      eqn = paste0(eqn, term, " + ")
    } else {
      eqn = paste0(eqn, term)
    }
  }
  return(eqn)
}

fr_n = function(x, r, n, a, b) {
  pdf_x = dunif(x, min=a, max=b)
  cdf_x = punif(x, min=a, max=b)
  numerator = factorial(n)/(factorial(r-1)*factorial(n-r))
  
  prob_dist = numerator*cdf_x^(r-1)*(1-cdf_x)^(n-r)*pdf_x
  return(prob_dist)
}

fr_n_eqn = function(r, n) {
  numerator = factorial(n) / (factorial(r-1)*factorial(n-r))
  prob_eqn = paste0(numerator, " F(x)^", r-1, " * [1 - F(x)]^", n-r, " * f(x)")
  return(prob_eqn)
}

moment_k = function(k, r, n, a, b){
  integrand = function(x) x^k * fr_n(x, r, n, a, b)
  moment = integrate(integrand, a, b)$value
  
  return(moment)
}

exp_xy = function(n, a, b) {
  integrand = function(x) {
    f_y = function(y) x * y * n * (n-1) * (y-x)
    integrate(Vectorize(f_y), lower = x, upper =b)$value
  }
  moment = integrate(Vectorize(integrand), a, b)$value
  return(moment)
}
